mission "R&R Pirate Reputation Monitor"
	invisible
	repeat
	to offer
		"reputation: Pirate" >= 100
		"reputation: R&R Stranded Ships Operations" >= 0
	on offer
		"reputation: R&R Stranded Ships Operations" = -10

mission "R&R Mid Tier Requirement Monitor"
	invisible
	repeat
	to offer
		or
			and
				"rr_operations_jobs_done" >= 20
				"combat rating" >= 403
			"rr_operations_jobs_done" >= 40
	on offer
		"rr_mid_tier_req" = 1

mission "R&R Hard Tier Requirement Monitor"
	invisible
	repeat
	to offer
		"rr_mid_tier_unlocked" == 1
		or
			and
				"rr_operations_jobs_done" >= 50
				"combat rating" >= 8103
			"rr_operations_jobs_done" >= 100
	on offer
		"rr_hard_tier_req" = 1

mission "R&R Milestone Reward 1"
	invisible
	to offer
		"rr_operations_jobs_done" >= 10

	on offer
		conversation
			"Dana's face appears on your terminal as soon as you dock. 'I was just reviewing our ledgers, Captain. You've successfully pulled off 10 operations for us now. We appreciate reliability around here, so I've authorized a bonus to your account. Keep up the good work.'"

	on complete
		payment 100000
