# Republic Recovery and Rescue jobs
# Easy ones

mission "Recovery Standard Republic Asset"
	name "Recover Republic Railgun Prototype in <waypoints>"
	description "A forward Naval patrol ship carrying a high-velocity 'Experimental Railgun Housing' has been disabled in the <waypoints> system."
	repeat
	source
		attributes "human"
	destination <origin>
	waypoint 
		distance 2 7
	deadline
	to offer
		"rr_intro_completed" == 1
	npc assist
		system <waypoints>
		government "Derelict"
		personality derelict target pacifist mute
		ship "Headhunter" "Stranded Navy Vessel"
		on board
			conversation
				"Your crew cuts open the forward weapon bay extracting the dense magnetic rails. The engineer radios that you have the hardware but slicing the magnetic lock tripped an automated thermal overload urging you to get out of there."
	npc
		system <waypoints>
		government "Pirate"
		personality staying
		fleet "Small Southern Pirates" 2
	on complete
		conversation
			"You return to <origin>. Fleet ordnance officers assume control of the weapon asset."
		"rr_operations_jobs_done" ++
		"reputation: Republic" += 0.01
		payment 45000
	on fail
		conversation
			"The Navy is furious that you allowed advanced Republic ballistics technology to be lost."
		"reputation: Republic" -= 0.15

	source
		attributes "human"
	destination <origin>
	waypoint 
		distance 8 13
	deadline
	to offer

	npc assist
		system <waypoints>
		government "Derelict"
		personality derelict target pacifist mute
		ship "Leviathan" "Burning Fleet Flagship"
		on board
			conversation
				"Your plasma-cutting frames extract the massive humming magnetic containment vessel. The deck chief yells that the antimatter core is secure in your bay but severing the stabilizing mounts has started a total structural explosion cycle urging you to disengage as heavy raider warships and swarms of fast interceptors drop directly out of hyperspace on your coordinate tracks."
			"republic_endgame_salvage" = 1
	npc
		system <waypoints>
		government "Pirate"
		personality staying
		fleet "Small Southern Pirates" 3
		fleet "Small Core Pirates" 2
		fleet "Large Core Pirates" 1
		fleet "Large Southern Pirates" 1
	on complete
		conversation
