# Republic Recovery and Rescue jobs

# Easy ones
mission "Recovery Standard Republic Asset"
	name "Recover Republic Railgun Prototype in <destination>"
	description "The Republic Navy has contracted you to cut open the forward weapon bay of a stranded vessel in the <destination> and extract its dense magnetic rails weighting <cargo>. Drop the prototype off at the local spaceport there. Payment for returning the prototype is 45,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 2 7
		not attributes "uninhabited"
	cargo 10 20 0.7
	deadline
	to offer
		"rr_intro_completed" == 1

	npc
		system destination
		government "Derelict"
		personality "derelict" "staying"
		ship "Scout" "Stranded Navy Vessel"
		on assist
			"rep_asset_ambush_1_ready" = 1
			conversation
				"Your crew cuts open the forward weapon bay extracting the dense magnetic rails."

	on complete
		conversation
			"You touch down at the spaceport where a crisp, uniformed Republic Navy quartermaster is waiting. They inspect the magnetic rails with an approving nod. 'Efficient work, Captain. The Navy relies on contractors who can execute orders without unnecessary complications. Your payment has been authorized.'"
		"rr_operations_jobs_done" ++
		"reputation: Republic" += 0.01
		"reputation: R&R Stranded Ships Operations" += 0.01
		payment 45000
	on fail
		conversation
			"Republic Navy quartermasters note a significant failure on your operational record."
		"reputation: Republic" -= 0.15
		"reputation: R&R Stranded Ships Operations" -= 0.15
	on abort
		conversation
			"A Republic logistics clerk looks at your file with indifference. 'Aborting a standard recovery mission? We don't have time for hesitation. Your lack of professionalism has been noted and marked in your personnel file.'"
		"reputation: Republic" -= 0.20
		"reputation: R&R Stranded Ships Operations" -= 0.20

# Hard ones
mission "Recovery Endgame Republic Core"
	name "CRITICAL: Extract Republic Antimatter Engine from <destination>"
	description "CRITICAL CONTRACT: Use plasma-cutting frames to extract a massive, humming antimatter engine core weighting <cargo> from a burning Fleet Flagship in the <destination>. Deliver it to the local spaceport there. Discretion and combat efficiency are mandatory. Payment for this endgame salvage operation is 250,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 8 13
		not attributes "uninhabited"
	cargo 110 150 0.15
	deadline
	to offer
		"your_mission_cooldown_date" <= day
		"rr_hard_tier_unlocked" == 1

	npc
		system destination
		government "Derelict"
		personality "derelict" "staying"
		ship "Container Transport" "Burning Fleet Flagship"
		on assist
			"rep_endgame_asset_ambush_ready" = 1
			conversation
				"Your team successfully cuts the antimatter engine core free as the flagship's hull integrity drops to critical levels."

	on complete
		conversation
			"You carefully touch down at the spaceport, engines whining from the strain. Republic Fleet engineers in hazard suits quickly swarm the hold, securing the humming antimatter core. A high-ranking Admiral contacts you directly. 'Securing that core denied our enemies a massive tactical advantage. Your operational excellence has been permanently noted by the Navy brass.'"
		"rr_operations_jobs_done" ++
		"rr_hard_tier_jobs_done" ++
		"reputation: Republic" += 0.15
		"reputation: R&R Stranded Ships Operations" += 0.15
		"your_mission_cooldown_date" = day + 15
		payment 250000
	on fail
		conversation
			"The Republic Navy officially terminates your clearance for the loss of a top-tier Fleet asset."
		"reputation: Republic" -= 0.40
		"reputation: R&R Stranded Ships Operations" -= 0.40
		"your_mission_cooldown_date" = day + 30
	on abort
		conversation
			"The Republic Admiral's voice is frigid. 'That antimatter core was a strategic necessity. Your choice to abort has severely compromised our local defensive capabilities. Your clearance is revoked—immediately.'"
		"reputation: Republic" -= 0.50
		"reputation: R&R Stranded Ships Operations" -= 0.50
		"your_mission_cooldown_date" = day + 30
