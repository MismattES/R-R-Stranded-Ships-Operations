# Free Worlds jobs

# Easy ones
mission "Recovery Standard Free Worlds Asset"
	name "Salvage Free Worlds Frontier Mineral Cache in <destination>"
	description "Free Worlds quartermasters need a ship to salvage a dense containment crate weighting <cargo> from a battered frontier miner in the <destination>. Drop the cache off at the local spaceport there. Payment is 100,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 2 7
		not attributes "uninhabited"
	cargo 10 20 0.7
	deadline
	to offer
		has "event: start of hostilities"
		"rr_intro_completed" == 1

	npc
		system destination
		government "Derelict"
		personality "derelict" "staying"
		ship "Scout" "Battered Frontier Miner"
		on assist
			"fw_asset_ambush_1_ready" = 1
			conversation
				"Your crew winches the dense heavy containment crate into your primary loading bay."
	on complete
		conversation
			"You land at the local spaceport to the sight of an overworked Free Worlds quartermaster wiping sweat from their brow. They practically run onto your ship to verify the containment crate. 'Thank the stars you pulled this off,' they say with a tired grin. 'We were days away from a critical shortage. Good flying, pilot.'"
		"rr_operations_jobs_done" ++
		"reputation: Free Worlds" += 0.01
		"reputation: R&R Stranded Ships Operations" += 0.01
		payment 100000
	on fail
		conversation
			"The Free Worlds militia command expresses their disappointment regarding the botched frontier recovery operation."
		"reputation: Free Worlds" -= 0.15
		"reputation: R&R Stranded Ships Operations" -= 0.15
	on abort
		conversation
			"The Free Worlds quartermaster shakes his head. 'We needed that cache, and you abandoned your post. I didn't think you were the type to turn tail when things got tough. You've lost our trust.'"
		"reputation: Free Worlds" -= 0.20
		"reputation: R&R Stranded Ships Operations" -= 0.20

mission "Recovery Endgame Free Worlds Crystals"
	name "CRITICAL: Salvage Free Worlds Plasma Cache in <destination>"
	description "CRITICAL CONTRACT: Extract a massive, highly volatile crystal cache weighting <cargo> from a crippled dreadnought in the <destination>. Deliver it to the local spaceport there. Expect overwhelming pirate resistance. Payment for this endgame salvage operation is 375,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 8 13
		not attributes "uninhabited"
	cargo 110 150 0.15
	deadline
	to offer
		"your_mission_cooldown_date" <= day
		"rr_hard_tier_unlocked" == 1

	npc
		system destination
		government "Derelict"
		personality "derelict" "staying"
		ship "Dreadnought" "Crippled Free Worlds Dreadnought"
		on assist
			"fw_endgame_asset_ambush_ready" = 1
			conversation
				"Your crew successfully extracts the massive highly volatile crystal crate."

	on complete
		conversation
			"You safely navigate down to the spaceport, where specialized Free Worlds hazard teams are waiting with heavy lifting rigs. As they carefully extract the highly volatile crystal cache, the local foreman shakes your hand vigorously. 'This plasma cache stabilizes the regional power grid for months, pilot. You didn't just salvage a wreck today; you kept the lights on for thousands.'"
		"rr_operations_jobs_done" ++
		"rr_hard_tier_jobs_done" ++
		"reputation: Free Worlds" += 0.15
		"reputation: R&R Stranded Ships Operations" += 0.15
		"your_mission_cooldown_date" = day + 15
		payment 375000
	on fail
		conversation
			"The Free Worlds command officer sighs, the transmission crackling with static. 'That asset was vital to our defense of the frontier, pilot. You've failed us when we needed you most. You are grounded until further notice while we re-evaluate your clearance.'"
		"reputation: Free Worlds" -= 0.40
		"reputation: R&R Stranded Ships Operations" -= 0.40
		"your_mission_cooldown_date" = day + 30
	on abort
		conversation
			"The quartermaster is practically shouting. 'That plasma cache was the only thing keeping the lights on in this sector! You’ve doomed thousands to darkness because you were afraid of a few pirates. Get out of our sight.'"
		"reputation: Free Worlds" -= 0.50
		"reputation: R&R Stranded Ships Operations" -= 0.50
		"your_mission_cooldown_date" = day + 30
