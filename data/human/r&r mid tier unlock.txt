mission "R&R Mid-Tier Invitation"
	name "Freelancer Contract: Mid-Tier Qualification"
	description "Dana is testing your capabilities. Act as the primary extraction unit for a hull stranded in the Regulus system. Be prepared for heavy hostile interception. Return to Ada to complete the contract and receive your clearance."
	source
		planet "Ada"
	destination "Ada"
	waypoint "Regulus"
	to offer
		"rr_mid_tier_unlocked" == 0
		"rr_mid_tier_started" == 0
		"rr_mid_tier_cooldown" <= day
		"rr_mid_tier_req" = 1

	on offer
		conversation
			"Dana's face appears on your terminal. She tells you she has been tracking your work, and you are ready for more. She has a critical rescue operation in the Regulus system, and if you pull it off, she will clear you for all of their high-priority contracts."
			choice
				"Send the coordinates. I'm on my way."
					accept
				"Not right now."
					decline

	on accept
		"rr_mid_tier_started" = 1

	passengers 20 10 1

	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Falcon" "Damaged R&R Ops Recovery Hull"
		on assist
			"rr_mid_tier_ambush_ready" = 1
			conversation
				"You're the extraction unit? Good, we've got heavy hostiles on our tail!"

	on complete
		conversation
			"Excellent work. You've proven you can handle the heat. I'm updating your clearance level now."
		"rr_mid_tier_unlocked" = 1
		"rr_mid_tier_started" = 0
		"reputation: R&R Stranded Ships Operations" += 0.10

	on fail
		conversation
			"Dana's expression is cold. 'The operation was a disaster. We can't afford these kinds of losses on a mid-tier contract. I'm locking you out of R&R field operations for a while so you can reflect on your performance.'"
		"rr_mid_tier_started" = 0
		"reputation: R&R Stranded Ships Operations" -= 0.20
		"rr_mid_tier_cooldown" = day + 15

mission "R&R Mid-Tier Ambush"
	invisible
	to offer
		"rr_mid_tier_ambush_ready" == 1
	deadline 1
	npc
		government "Pirate"
		personality "hunting"
		fleet "Small Southern Pirates" 2
