mission "R&R Operations Introduction"
	name "Freelancer Contract: Simple Retrieval"
	description "Dana, a representative of a specialized recovery organization, has contracted you for a retrieval in the Eteron system. Return to Mars to receive your 25,000 credits."
	source "Mars"
	destination "Mars"
	waypoint "Eteron"
	to offer
		"rr_intro_completed" == 0
		"rr_intro_cooldown" <= day

	on offer
		conversation
			branch "branch_a"
				"rr_intro_offered" == 0
				"rr_intro_failed_once" == 1
			branch "branch_b"
				"rr_intro_offered" == -1
			"Hey, my name is Dana. I am a member of an organization that deals in recovering things. What do you think about doing a little retrieval in Eteron?"
			goto "offer_choice"
			label "branch_a"
				"Let's try this again, pilot."
			goto "offer_choice"
			label "branch_b"
				"Are you ready to take the contract now?"
			label "offer_choice"
			choice
				"I can handle that. Lock in the coordinates."
					accept
				"Not interested."
					decline

	on accept
		"rr_intro_offered" = 1
		"rr_intro_completed" = 0

	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Headhunter" "Disabled Auxiliary Hull"
		on assist
			"rr_intro_ambush_ready" = 1
			conversation
				"Your engineering crew managed to get the system requested. Time to go find Dana."

	on complete
		conversation
			"Dana is waiting for you at the landing pad. When you give her the box, she smiles and informs you that she has more work for you."
		"rr_intro_completed" = 1
		"reputation: R&R Stranded Ships Operations" += 0.02
		"rr_operations_jobs_done" ++
	on fail
		conversation
			"Dana looks at you with an annoyed face and informs you that you still need to get more experience before getting another shot. You feel a rush of shame watching her go away."
		"rr_intro_cooldown" = day + 30
		"reputation: R&R Stranded Ships Operations" -= 0.15
