# Syndicate Recovery and Rescue jobs
# Easy ones

mission "Recovery Standard Syndicate Asset"
	name "Recover Syndicate Cybernetic Core from <waypoints>"
	description "A corporate transport carrying a highly sensitive 'Stolen Cybernetic Server Array' has been ambushed in the <waypoints> system."
	repeat
	source
		government "Syndicate"
	destination <origin>
	waypoint 
		distance 2 7
	deadline
	to offer
		"rr_intro_completed" == 1
	npc assist
		system <waypoints>
		government "Derelict"
		personality derelict target pacifist mute
		ship "Headhunter" "Disabled Syndicate Asset"
		on board
			conversation
				"Your team breaks the terminal locks disconnecting the liquid-cooled processing stack. The hacker reports the core is detached but warns that the ship's security software just triggered an engine self-destruct cycle and you need to move."
	npc
		system <waypoints>
		government "Pirate"
		personality staying
		fleet "Small Southern Pirates" 2
	on complete
		conversation
			"You return to <origin>. Syndicate technicians quickly wheel away the data array under heavy encryption covers and your holds are clear."
		"rr_operations_jobs_done" ++
		"reputation: Syndicate" += 0.01
		payment 45000
	on fail
		conversation
			"The corporate board files legal penalties against your profile for allowing proprietary software assets to be wiped."
		"reputation: Syndicate" -= 0.15
