# Syndicate Recovery and Rescue jobs

# Easy ones
mission "Recovery Standard Syndicate Asset"
	name "Recover Syndicate Cybernetic Core from <destination>"
	description "Syndicate operatives require an independent contractor to break terminal locks and disconnect a liquid-cooled processing stack weighting <cargo> from a disabled asset in the <destination>. Drop the core off at the local spaceport there. Payment for returning the cybernetic core is 100,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 2 7
		not attributes "uninhabited"
	cargo 10 20 0.7
	deadline
	to offer
		"rr_intro_completed" == 1

	npc
		system destination
		government "Derelict"
		personality "derelict" "staying"
		ship "Clipper" "Disabled Syndicate Asset"
		on assist
			"syn_asset_ambush_1_ready" = 1
			conversation
				"Your team breaks the terminal locks disconnecting the liquid-cooled processing stack."

	on complete
		conversation
			"You land at the local spaceport, where a pair of Syndicate operatives in tailored suits are waiting. They load the liquid-cooled processing stack onto a sleek hover-cart without a single word. One of them slides a cred-stick into your hand and smirks. 'Clean, quiet, and on time. Exactly how the board likes it.'"
		"rr_operations_jobs_done" ++
		"reputation: Syndicate" += 0.01
		"reputation: R&R Stranded Ships Operations" += 0.01
		payment 100000
	on fail
		conversation
			"A Syndicate clerk efficiently logs your failure, noting a minor deduction in your reliability rating."
		"reputation: Syndicate" -= 0.15
		"reputation: R&R Stranded Ships Operations" -= 0.15
	on abort
		conversation
			"A Syndicate clerk looks up from their terminal with utter boredom. 'Quitting a job mid-operation? That's a breach of contract, Captain. Our clients don't like paying for hesitation. Consider your standing with us permanently tarnished.'"
		"reputation: Syndicate" -= 0.20
		"reputation: R&R Stranded Ships Operations" -= 0.20

mission "Recovery Endgame Syndicate Tech"
	name "CRITICAL: Salvage Syndicate AI Logic Core in <destination>"
	description "CRITICAL CONTRACT: Override the cryogenic locks on a server farm weighting <cargo> aboard a ruined corporate super-freighter in the <destination>. Deliver it to the local spaceport there. The Syndicate expects flawless execution and for you to ask no questions regarding the AI Logic Core. Payment for this endgame salvage operation is 375,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 8 13
		not attributes "uninhabited"
	cargo 110 150 0.15
	deadline
	to offer
		"your_mission_cooldown_date" <= day
		"rr_hard_tier_unlocked" == 1

	npc
		system destination
		government "Derelict"
		personality "derelict" "staying"
		ship "Hauler II" "Ruined Corporate Super-Freighter"
		on assist
			"syn_endgame_asset_ambush_ready" = 1
			conversation
				"Your specialist team overrides the cryogenic locks on the server farm."

	on complete
		conversation
			"You safely navigate down to the spaceport, greeted by a phalanx of corporate executives supervising the extraction of the server farm. A senior supervisor hands you a hefty cred-stick. 'Flawless execution under heavy fire,' he notes, eyes gleaming. 'The Syndicate appreciates assets who deliver priceless tech without asking dangerous questions.'"
		"rr_operations_jobs_done" ++
		"rr_hard_tier_jobs_done" ++
		"reputation: Syndicate" += 0.15
		"reputation: R&R Stranded Ships Operations" += 0.15
		"your_mission_cooldown_date" = day + 15
		payment 375000
	on fail
		conversation
			"A Syndicate black-suit supervisor leans into the camera, his voice a low, dangerous whisper. 'You were hired because we thought you were efficient. Losing that AI core has cost the Syndicate millions. Consider yourself suspended from all further operations while we settle our accounts.'"
		"reputation: Syndicate" -= 0.40
		"reputation: R&R Stranded Ships Operations" -= 0.40
		"your_mission_cooldown_date" = day + 30
	on abort
		conversation
			"A black-suit supervisor leans into the frame, voice a low, lethal whisper. 'That core was worth millions. You’ve cost us more than just time; you’ve cost us leverage. You are finished in Syndicate airspace.'"
		"reputation: Syndicate" -= 0.50
		"reputation: R&R Stranded Ships Operations" -= 0.50
		"your_mission_cooldown_date" = day + 30
