mission "Recovery Standard Free Worlds Asset"
	name "Salvage Free Worlds Frontier Mineral Cache in <waypoints>"
	description "A local mining courier carrying a heavy load of 'Refined Hyper-Density Magnetite' has been stranded in the <waypoints> system."
	repeat
	source
		government "Free Worlds"
	destination <origin>
	waypoint 
		distance 2 7
	deadline
	to offer
		"rr_intro_completed" == 1
	npc assist
		system <waypoints>
		government "Derelict"
		personality derelict target pacifist mute
		ship "Bounder" "Battered Frontier Miner"
		on board
			conversation
				"Your crew winches the dense heavy containment crate into your primary loading bay. The deckhands yell that the cargo is secure but the structural framework is buckling from the shifting weight and urge you to clear the docking clamps."
	npc
		system <waypoints>
		government "Pirate"
		personality staying
		fleet "Small Southern Pirates" 2
	on complete
		conversation
			"You touch down at <origin>. Local factory builders eagerly offload the critical raw manufacturing minerals and your holds are clear."
		"rr_operations_jobs_done" ++
		"reputation: Free Worlds" += 0.01
		payment 45000
	on fail
		conversation
			"The Free Worlds Council laments the loss of critical war production metals to regional marauders."
		"reputation: Free Worlds" -= 0.15
